            }
        }

        case "LOGIN_USER_REJECTED": {
            return {
                ...state,
                fetching: false,
                isLoggedIn: false,
                error: action.payload
            }
        }

        case "LOGOUT_USER": {
            return {
                ...state,
                fetching: true,
                fetched: false,
            }
        }

        case "LOGOUT_USER_FULFILLED": {
            return{

                fetched: true,
                user: null,

        }

        case "LOGOUT_USER_REJECTED": {
            return {
                ...state,
                fetching: false,

                error: action.payload
            }
        }

    }
